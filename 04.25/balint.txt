SELECT v.vnev AS VersenyzoNeve, v.szul_ev AS SzuletesiIdeje
FROM versenyzok v
JOIN merkozesek m ON v.versenyzoid = m.vnevid1 OR v.versenyzoid = m.vnevid2
JOIN sulycsoport s ON m.sulycsopid = s.id
WHERE s.sulycsoport >= '91+' AND m.datum = '2023'
AND v.nem = 'ferfi'
AND szul_ev IS NOT NULL
ORDER BY v.szul_ev ASC
LIMIT 1;

SELECT versenyzok.vnev , COUNT(merkozesek.KO)
FROM merkozesek
JOIN versenyzok ON  versenyzok.versenyzoid=merkozesek.vnevid1  OR versenyzok.versenyzoid=merkozesek.vnevid2 
JOIN sulycsoport ON merkozesek.sulycsopid = sulycsoport.id
WHERE versenyzok.nem = 'ferfi' AND sulycsoport.sulycsoport = 'Kislégsúly' OR sulycsoport.sulycsoport = 'Szupernehézsúly'
AND merkozesek.KO=1
AND merkozesek.gyoztesid=versenyzok.versenyzoid
GROUP BY merkozesek.gyoztesid
HAVING COUNT(merkozesek.KO)=1;

SELECT versenyzok.nemzetiseg AS Ország,
       COUNT(DISTINCT CASE WHEN datum = 2023 THEN versenyzok.versenyzoid END) AS '2023-as indulók száma',
       COUNT(DISTINCT CASE WHEN datum = 2024 THEN versenyzok.versenyzoid END) AS '2024-es indulók száma',
       COUNT(DISTINCT versenyzok.versenyzoid) AS 'összes induló'
FROM versenyzok
JOIN merkozesek ON versenyzok.versenyzoid = merkozesek.vnevid1 OR versenyzok.versenyzoid = merkozesek.vnevid2
GROUP BY versenyzok.nemzetiseg
ORDER BY COUNT(DISTINCT versenyzok.versenyzoid) DESC;

FROM merkozesek
