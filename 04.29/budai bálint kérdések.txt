FROM termekek

Melyik kategóriában található a legtöbb termék?
SELECT COUNT(termId), KatNev
FROM termekek,kategoriak,alkategoria
WHERE termekek.alkatId=alkategoria.alkatId AND kategoriak.katId=alkategoria.katId
GROUP BY kategoriak.katId
ORDER BY kategoriak.katId

Mennyi az összes rendelés összértéke? 
SELECT SUM(termAr*db) "összeg"
FROM termekek,tetel
WHERE termekek.termId=tetel.termId
