SELECT versenyzok.vnev,merkozesek.KO
FROM merkozesek
JOIN versenyzok ON merkozesek.vnevid1=versenyzok.versenyzoid OR merkozesek.vnevid2=versenyzok.versenyzoid
WHERE versenyzok.versenyzoid=merkozesek.gyoztesid
AND versenyzok.nem='no'
AND merkozesek.KO=1
